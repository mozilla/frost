    """

[docs]def is_owner_of_group(member):
    """
    Check whether a member of a group is an owner with a status of 'ACTIVE'.
    """
    return (
        member["type"] == "USER"
        and member["role"] == "OWNER"
        and member["status"] == "ACTIVE"
    )
